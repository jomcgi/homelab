# Bazel Bootstrap Guide for Homelab Monorepo

**Purpose:** Comprehensive guide for setting up and maintaining Bazel in this personal monorepo with enterprise-grade patterns.

**Philosophy:** Treat personal projects as platform engineering practice. Zero-touch CI, self-service onboarding, centralized tooling.

## Phase 2: Platform Rules (Day 1-2, ~4 hours)

### `build_defs/go.bzl`

```python
"""Standard Go service patterns."""

load("@rules_go//go:def.bzl", "go_binary", "go_library", "go_test")
load("@rules_oci//oci:defs.bzl", "oci_image", "oci_push", "oci_image_index")
load("@rules_pkg//pkg:tar.bzl", "pkg_tar")

def go_service(
        name,
        srcs = None,
        deps = [],
        embed = [],
        data = [],
        visibility = None,
        image_registry = "ghcr.io/jomcgi",
        **kwargs):
    """
    Standard Go microservice with binary, image, and push targets.
    
    Creates:
    - :{name}           - go_binary
    - :{name}_image     - oci_image (single platform)
    - :{name}_index     - oci_image_index (multi-platform)
    - :{name}_push      - oci_push to registry
    
    Usage:
        go_service(
            name = "my-service",
            deps = ["//pkg/common"],
        )
    
    Then: bazel build :my-service_image
          bazel run :my-service_push
    """
    
    # Binary
    go_binary(
        name = name,
        srcs = srcs,
        deps = deps,
        embed = embed,
        data = data,
        visibility = visibility or ["//visibility:public"],
        **kwargs
    )
    
    # Package binary into tar for OCI layer
    pkg_tar(
        name = name + "_layer",
        srcs = [":" + name],
        package_dir = "/app",
    )
    
    # OCI Image (single platform for local dev)
    oci_image(
        name = name + "_image",
        base = "@distroless_base",
        entrypoint = ["/app/" + name],
        tars = [":" + name + "_layer"],
        labels = _standard_labels(),
        user = "nonroot",
        workdir = "/app",
    )
    
    # Multi-platform image index (for production)
    oci_image_index(
        name = name + "_index",
        images = [
            ":" + name + "_image",
        ],
    )
    
    # Push target
    oci_push(
        name = name + "_push",
        image = ":" + name + "_index",
        repository = image_registry + "/" + name,
    )

def go_library_with_tests(
        name,
        srcs,
        deps = [],
        test_deps = [],
        **kwargs):
    """
    Go library with automatic test discovery.
    
    Creates library + test target if *_test.go files exist.
    """
    
    go_library(
        name = name,
        srcs = [s for s in srcs if not s.endswith("_test.go")],
        deps = deps,
        **kwargs
    )
    
    test_srcs = [s for s in srcs if s.endswith("_test.go")]
    if test_srcs:
        go_test(
            name = name + "_test",
            srcs = test_srcs,
            embed = [":" + name],
            deps = deps + test_deps + [
                "@com_github_stretchr_testify//assert",
                "@com_github_stretchr_testify//require",
            ],
        )

def _standard_labels():
    """Standard OCI labels for all images."""
    return {
        "org.opencontainers.image.source": "https://github.com/jomcgi/homelab",
        "org.opencontainers.image.licenses": "MIT",
        "org.opencontainers.image.vendor": "jomcgi",
    }
```

### `build_defs/python.bzl`

```python
"""Standard Python service patterns."""

load("@rules_python//python:defs.bzl", "py_binary", "py_library", "py_test")
load("@rules_oci//oci:defs.bzl", "oci_image", "oci_push")
load("@rules_pkg//pkg:tar.bzl", "pkg_tar")

def py_service(
        name,
        srcs,
        deps = [],
        main = "main.py",
        data = [],
        visibility = None,
        image_registry = "ghcr.io/jomcgi",
        **kwargs):
    """
    Standard Python service (FastAPI/Flask pattern).
    
    Creates:
    - :{name}           - py_binary
    - :{name}_image     - oci_image
    - :{name}_push      - oci_push to registry
    """
    
    # Library for reusable code
    py_library(
        name = name + "_lib",
        srcs = [s for s in srcs if s != main],
        deps = deps + [
            "@pypi//fastapi",
            "@pypi//uvicorn",
            "@pypi//pydantic",
        ],
        visibility = ["//visibility:private"],
    )
    
    # Binary
    py_binary(
        name = name,
        srcs = [main],
        deps = [":" + name + "_lib"],
        data = data,
        main = main,
        visibility = visibility or ["//visibility:public"],
        **kwargs
    )
    
    # Package for OCI
    pkg_tar(
        name = name + "_layer",
        srcs = [":" + name],
        package_dir = "/app",
    )
    
    # OCI Image
    oci_image(
        name = name + "_image",
        base = "@distroless_python",
        entrypoint = ["python", "/app/" + main],
        tars = [":" + name + "_layer"],
        user = "nonroot",
    )
    
    # Push
    oci_push(
        name = name + "_push",
        image = ":" + name + "_image",
        repository = image_registry + "/" + name,
    )

def py_test_suite(name, srcs, deps = [], **kwargs):
    """Standard pytest suite."""
    py_test(
        name = name,
        srcs = srcs,
        deps = deps + [
            "@pypi//pytest",
        ],
        main = srcs[0],
        **kwargs
    )
```

## Phase 4: CI Configuration (Day 2, ~2 hours)

### `.github/workflows/ci.yml`

```yaml
name: CI
on:
  pull_request:
  push:
    branches: [main]

jobs:
  test:
    name: Test affected targets
    runs-on: ubuntu-latest-8-cores  # Use larger runners for Bazel
    
    steps:
      - name: Checkout
        uses: actions/checkout@v4
        with:
          fetch-depth: 0  # Need history for diffs
      
      - name: Setup Bazel
        uses: bazel-contrib/setup-bazel@0.8.1
        with:
          bazelisk-version: "1.19.0"
          bazelisk-cache: true
          repository-cache: true
          disk-cache: false
      
      - name: Configure BuildBuddy
        run: |
          cat >> .bazelrc.user << EOF
          build --config=ci
          build --remote_header=x-buildbuddy-api-key=${{ secrets.BUILDBUDDY_API_KEY }}
          EOF
      
      - name: Compute affected targets
        id: affected
        run: |
          if [ "${{ github.event_name }}" = "pull_request" ]; then
            BASE="origin/${{ github.base_ref }}"
            CHANGED=$(git diff --name-only $BASE..HEAD | tr '\n' ' ')
            TARGETS=$(bazel query "tests(rdeps(//..., set($CHANGED)))" 2>/dev/null || echo "//...")
          else
            TARGETS="//..."
          fi
          echo "targets=$TARGETS" >> $GITHUB_OUTPUT
      
      - name: Run tests
        run: |
          bazel test ${{ steps.affected.outputs.targets }} \
            --test_output=errors \
            --build_metadata=ROLE=CI \
            --build_metadata=COMMIT_SHA=${{ github.sha }} \
            --build_metadata=REPO_URL=https://github.com/${{ github.repository }}
      
      - name: Build all targets
        run: bazel build //...

  lint:
    name: Lint
    runs-on: ubuntu-latest
    
    steps:
      - uses: actions/checkout@v4
      - uses: bazel-contrib/setup-bazel@0.8.1
      
      - name: Buildifier
        run: |
          bazel run //:buildifier_test
      
      - name: Gazelle check
        run: |
          bazel run //:gazelle
          if ! git diff --quiet; then
            echo "BUILD files out of date. Run: bazel run //:gazelle"
            git diff
            exit 1
          fi

  images:
    name: Build and push images
    runs-on: ubuntu-latest
    needs: [test, lint]
    if: github.ref == 'refs/heads/main'
    
    permissions:
      contents: read
      packages: write
    
    steps:
      - uses: actions/checkout@v4
      - uses: bazel-contrib/setup-bazel@0.8.1
      
      - name: Login to GHCR
        uses: docker/login-action@v3
        with:
          registry: ghcr.io
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}
      
      - name: Build and push images
        run: |
          IMAGES=$(bazel query "kind('oci_push', //...)")
          for img in $IMAGES; do
            echo "Pushing $img"
            bazel run $img
          done
```

## Best Practices from Enterprise Experience

### 1. BUILD File Organization

```python
# Good: Clear sections
load("@rules_go//go:def.bzl", "go_library", "go_binary")
load("//build_defs:go.bzl", "go_service")

# Libraries first
go_library(
    name = "lib",
    srcs = glob(["*.go"], exclude=["*_test.go"]),
    deps = [...],
)

# Tests
go_test(
    name = "lib_test",
    srcs = glob(["*_test.go"]),
    embed = [":lib"],
)

# Binaries
go_service(
    name = "myservice",
    deps = [":lib"],
)
```

### 2. Dependency Management

```python
# Good: Use platform rules for consistency
load("//build_defs:go.bzl", "go_service")
go_service(name = "api")  # Gets standard deps automatically

# Avoid: Direct imports that bypass platform
go_binary(
    name = "api",
    deps = ["@go_sdk//..."],  # Don't do this
)
```

### 3. Visibility Control

```python
# Package-private by default
go_library(
    name = "internal",
    visibility = ["//visibility:private"],
)

# Explicit public
go_library(
    name = "api",
    visibility = ["//visibility:public"],
)

# Limited visibility
go_library(
    name = "shared",
    visibility = [
        "//services:__subpackages__",
        "//pkg:__subpackages__",
    ],
)
```

### 4. Testing Patterns

```python
# Fast unit tests (size = small, no external deps)
go_test(
    name = "unit_test",
    srcs = ["foo_test.go"],
    size = "small",
    tags = ["unit"],
)

# Integration tests (size = large, can use network/docker)
go_test(
    name = "integration_test",
    srcs = ["integration_test.go"],
    size = "large",
    tags = ["integration", "requires-docker"],
    data = ["//testdata:fixtures"],
)
```

### 5. Generated Code

```python
# Keep generated code in BUILD file, not Git
genrule(
    name = "generate_proto",
    srcs = ["api.proto"],
    outs = ["api.pb.go"],
    cmd = "protoc --go_out=$(@D) $(SRCS)",
)

# .gitignore
*.pb.go
*_generated.go
```

### 6. Cross-Language Dependencies

```python
# Python service depending on Go binary
py_binary(
    name = "orchestrator",
    srcs = ["main.py"],
    data = [
        "//services/worker:worker",  # Go binary as data file
    ],
```

## Troubleshooting Guide

### Problem: "ERROR: no such package"

```bash
# Check if BUILD file exists
ls path/to/package/BUILD.bazel

# Regenerate BUILD files
bazel run //:gazelle

# Check MODULE.bazel has correct dependencies
```

### Problem: "cannot find module providing package"

```bash
# Go dependency missing - update from go.mod
bazel run //:gazelle-update-repos

# Check MODULE.bazel has go_deps configured
```

### Problem: Slow builds

```bash
# Check remote cache is working
bazel build //... --execution_log_json_file=/tmp/exec.json
grep "cache_hit" /tmp/exec.json

# Profile the build
bazel build //... --profile=/tmp/profile.json
bazel analyze-profile /tmp/profile.json

# Check for unnecessary rebuilds
bazel build //... --explain=/tmp/explain.txt
```

### Problem: Tests failing in CI but not locally

```bash
# Run with same flags as CI
bazel test //... --config=ci

# Check for PATH differences
bazel test //... --incompatible_strict_action_env

# Run in hermetic mode
bazel test //... --sandbox_debug
```

### Problem: "permission denied" errors

```bash
# Sandbox issues - check file permissions
ls -la $(bazel info execution_root)

# Try disabling sandbox
bazel build //... --spawn_strategy=local

# macOS specific - grant Full Disk Access to terminal
```
