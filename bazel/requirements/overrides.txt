# Override lxml[html-clean] -> lxml to break circular dependency in Bazel.
#
# trafilatura depends on lxml[html-clean], which causes:
#   lxml -> lxml-html-clean -> lxml (cycle)
# Bazel's rules_python pip targets cannot handle dependency cycles.
#
# This override strips the [html-clean] extra. We add lxml-html-clean
# as an explicit dep where needed (extractors BUILD) so it's still
# available at runtime.
lxml==6.0.2

# Override pillow upper bound from open-gopro[gui] (requires <12).
# Pillow 12.1.1 is needed to fix CVE. The <12 bound in open-gopro
# is a cautious constraint, not a known incompatibility.
pillow>=12.1.1

# Override semgrep's tight compatible-release (~=) pins that conflict
# with runtime dependency versions. semgrep pins click~=8.1.8, rich~=13.5.2,
# exceptiongroup~=1.2.0, opentelemetry-sdk~=1.37.0, mcp==1.23.3, etc.,
# but runtime deps pin newer versions. Since all.in combines runtime and
# tools into a single pip.parse() lock file, we broaden these to let
# semgrep (a build-time-only tool) coexist with the runtime dependency set.
click>=8.1.8
rich>=13.5.2
exceptiongroup>=1.2.0
jsonschema>=4.25.1
mcp>=1.23.3
opentelemetry-api>=1.25.0
opentelemetry-sdk>=1.25.0
opentelemetry-exporter-otlp-proto-http>=1.25.0
